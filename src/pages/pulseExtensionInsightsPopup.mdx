import { PulseExtensionInsightsPopup } from 'components';
import { useSearchParams } from "next/navigation";


export default function InsightsPopupPage() {
  const searchParams = useSearchParams();
  let metricId = searchParams.get("metricId");
  if (process.env.DEBUG?.toLowerCase() === 'true') console.log(`metricId: ${metricId}`);
  return <PulseExtensionInsightsPopup metricId={metricId} />;
}
